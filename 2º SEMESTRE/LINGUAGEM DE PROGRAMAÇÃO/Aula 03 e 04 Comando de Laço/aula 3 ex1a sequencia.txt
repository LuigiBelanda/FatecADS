#include <stdio.h>
#include <stdlib.h>

int main() 
{
    int i,n,a,b;
    float s, sinal;
    printf("\n Entre com o numero de termos: ");
    scanf("%d",&n);
    a=1; b=3; s=0; sinal=1.0;
    printf("\n");
    for (i=1; i<=n; i++)
    {
        s= s + sinal*a/b;
        printf(" + %d/%d", a,b);
        a = (i+1)*(i+1);
        b = b + 4;
    }
    printf("\n s = %.2f", s);
    return 0;
}
